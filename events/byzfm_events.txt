byzfm.100 = { # Enables Imperial Palace excavation
	type = country_event
	
	title = byzfm.100.t
	desc = byzfm.100.d
	flavor = byzfm.100.f
	
	event_image = {
		video = "byzfm_ruins"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	duration = 3
	
	dlc = dlc_grefm
	
	trigger = {
		NOT = { has_variable = byzfm_palace_restoration }
		has_variable = byzfm_byzantium_formed
		owns_entire_state_region = STATE_EASTERN_THRACE
		is_subject = no
	}
	
	immediate = {
		s:STATE_EASTERN_THRACE = {
			random_scope_state = {
				save_scope_as = mega_palation_site
			}
		}
		set_variable = byzfm_palace_restoration
	}
	
	option = {
		name = byzfm.100.a
		default_option = yes
		add_journal_entry = {
			type = je_byzfm_mega_palation
		}
		custom_tooltip = {
			text = je_byzfm_mega_palation_tt
		}
	}
}
