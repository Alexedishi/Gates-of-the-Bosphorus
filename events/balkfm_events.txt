balkfm.030 = { # Creates Hersek with Smaïl-aga at the Helm
	type = country_event
	placement = root
	hidden = yes
	
	trigger = { # this is evaluated from MON scope
		NOT = { has_global_variable = balkfm_smail_aga_rebelled }
		year < 1940
		has_modifier = modifier_balkfm_tolerated_abuses
		c:TUR ?= {
			owns_entire_state_region = STATE_BOSNIA
			has_journal_entry = je_sick_man_main # serves as a time limit
		}
	}

	immediate = {
		set_global_variable = balkfm_smail_aga_rebelled
		random_character_in_void = {	
			limit = {
				has_variable = balkfm_is_smail_aga
			}
			save_scope_as = balkfm_smail_aga_scope
		}
		create_country = { # Hersek with Bosnia
			tag = HRSK
			origin = c:TUR
			state = s:STATE_BOSNIA.region_state:TUR
		}
		c:HRSK = {
			if = {
				limit = {
					exists = scope:balkfm_smail_aga_scope
				}
				scope:balkfm_smail_aga_scope = {
					free_character_from_void = yes
					set_character_as_ruler = yes
				}
			}
			else = {
				create_character = {
					# Smaïl-aga Čengić, Ottoman lord in Bosnia, killed on the orders of the Bishop
					first_name = Smail
					last_name = Cengic
					historical = yes
					age = 56 # born 1780
					culture = cu:serb
					religion = rel:sunni
					interest_group = ig_landowners
					ideology = ideology_moderate
					traits = {
						arrogant
						war_criminal
					}
					on_created = {
						set_variable = balkfm_is_smail_aga
					}
				}
			}
		}
		s:STATE_BOSNIA = {
			random_scope_state = {
				save_scope_as = bosnia_state_scope
			}
		}
		c:TUR = {
			trigger_event = {
				id = balkfm.337 # The Arrogance of Samïl-aga
			}
		}
		c:MON = {
			trigger_event = {
				id = balkfm.337 # The Arrogance of Samïl-aga
			}
			remove_modifier = modifier_balkfm_tolerated_abuses
		}
		every_country = {
			limit = {
				has_journal_entry = je_eqfm_sub
			}
			post_notification = notification_balkfm_hersek_smail
		}
	}
	
}
