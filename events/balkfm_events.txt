balkfm.100 = { # Great Eastern Crisis Warmup
	type = country_event
	placement = root
	
	title = balkfm.100.t
	desc = balkfm.100.d
	flavor = balkfm.100.f
	
	event_image = {
		video = "europenorthamerica_springtime_of_nations"
	}
	
	icon = "gfx/interface/icons/event_icons/event_ottoman.dds"
	
	on_opened_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	
	duration = 3
	
	dlc = dlc_grefm
	
	trigger = {
		NOT = { has_global_variable = balkfm_crisis_happened }
		NOT = { has_global_variable = balkfm_crisis_var }
		has_technology_researched = nationalism
		NOT = { has_technology_researched = pan-nationalism }
		is_subject = no
		is_junior_in_customs_union = no
		OR = {
			country_rank = rank_value:great_power
			country_rank = rank_value:major_power
			country_rank = rank_value:unrecognized_major_power
		}
		OR = {
			c:GRE ?= {
				has_technology_researched = nationalism
			}
			c:SER ?= {
				has_technology_researched = nationalism
			}
			c:MON ?= {
				has_technology_researched = nationalism
			}
		}
	}
	
	immediate = {
		c:RUS = {
			save_scope_as = russia_scope
		}
		set_global_variable = {
			name = balkfm_crisis_var
			value = 0
		}
		set_variable = balkfm_crisis_target
		set_variable = balkfm_crisis_offered
	}
	
	option = {
		name = balkfm.100.a
		default_option = yes
		add_journal_entry = {
			type = je_great_eastern_crisis_main
		}
		every_scope_state = {
			limit = {
				state_region = {
					NOT = {
						is_homeland = cu:turkish
					}
				}
				region = sr:region_balkans
			}
			add_radicals_in_state = {
				value = large_radicals
			}
		}
	}

}

				}

		}

		}

			c:PRU = {
				save_scope_as = germany_scope

					save_scope_as = germany_capital_scope
				}
			}
		}
	}

		trigger_event = {
			id = balkfm.110 # The Congress of Berlin

		}
	}

balkfm.108 = { # Creates GEC DP and Fires balkfm.109
	type = country_event
	placement = root
	hidden = yes
	
	trigger = {
		NOT = { has_global_variable = balkfm_crisis_happened }
		has_journal_entry = je_great_eastern_crisis_main
		global_var:balkfm_crisis_var >= 150
		is_diplomatic_play_committed_participant = no
		is_at_war = no
	}

	immediate = {
		set_global_variable = balkfm_crisis_happened
		if = {
			limit = {
				c:RUS = {
					has_variable = balkfm_crisis_attacker
				}
			}
			c:RUS = {
				set_variable = balkfm_crisis_starter
				save_scope_as = balkfm_crisis_starter_scope
			}
		}
		else = {			
			random_country = {
				limit = {
					has_variable = balkfm_crisis_attacker
					country_rank = rank_value:great_power
					is_diplomatic_play_committed_participant = no
				}
				set_variable = balkfm_crisis_starter
				save_scope_as = balkfm_crisis_starter_scope
			}
		}
		if = {
			limit = {
				exists = scope:balkfm_crisis_starter_scope
			}
			scope:balkfm_crisis_starter_scope = {
				create_diplomatic_play = {
					name = balkfm_great_eastern_crisis
					target_country = c:TUR
					type = dp_balkfm_crisis
					war = yes
				}
				trigger_event = {
					id = balkfm.104 # GEC Start (Initiator / Target)
					popup = yes
				}
			}
			trigger_event = {
				id = balkfm.109 # Assigns Tags and preps scopes for the event_window
				days = 1 # let it cook
			}
		}
		else = {
			remove_global_variable = balkfm_crisis_happened
		}
	}
	
}
