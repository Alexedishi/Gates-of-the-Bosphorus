byzfm_mega_palation_survey_button = {
	name = "byzfm_mega_palation_survey_button"
	desc = "byzfm_mega_palation_survey_button_desc"
	
	visible = {
		owner = {
			NOT = { has_variable = byzfm_mega_palation_surveyed }
		}
	}
	
	possible = {
		owns_entire_state_region = STATE_EASTERN_THRACE
		NOT = {
			has_journal_entry = je_byzfm_mega_palation_survey
		}
	}
	
	effect = {
		set_variable = { 
			name = byzfm_imperial_palace_var
			value = 0
		}
		add_journal_entry = {
			type = je_byzfm_mega_palation_survey
		}
	}
	
}

byzfm_mega_palation_construction_button = {
	name = "byzfm_mega_palation_construction_button"
	desc = "byzfm_mega_palation_construction_button_desc"
	
	visible = {
		NOT = {
			any_scope_state = {
				has_building = building_mega_palation
			}
		}
		has_variable = byzfm_mega_palation_surveyed
	}
	
	possible = {
		owns_entire_state_region = STATE_EASTERN_THRACE
		custom_tooltip = {
				text = balkfm_active_construction_tt
				hidden_trigger = {
					has_variable = balkfm_construction_var
				}
			}
		}
	}
	
	effect = {
		set_variable = { 
			name = balkfm_construction_var
			value = 0
		}
		add_journal_entry = {
			type = je_byzfm_mega_palation
		}
		add_modifier = modifier_balkfm_construction_1
	}
	
}
