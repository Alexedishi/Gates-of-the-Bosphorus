je_byzfm_mega_palation = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"
	
	group = je_group_byzfm_restoration

	immediate = {
	}

	complete = {
		any_scope_state = {
			has_building = building_mega_palation
		}
	}

	fail = {
		NOT = {
			owns_entire_state_region = STATE_EASTERN_THRACE
		}
	}

	on_complete = {
		trigger_event = { 
			id = byzfm.102 # Great Palace of Constantinople (palace rebuilt)
		}
	}
	
	scripted_button = byzfm_archeaology_mega_palation_button

	weight = 500
	
}


je_byzfm_mega_palation_survey = {
	icon = "gfx/interface/icons/event_icons/event_ruins.dds"
	
	group = je_group_byzfm_content

	on_monthly_pulse = {
		effect = {
			change_variable = {
				name = byzfm_imperial_palace_var
				add = 1
			}
		}
	}

	complete = {
		scope:journal_entry = { is_goal_complete = yes }
	}

	on_complete = {
		remove_modifier = modifier_grefm_active_excavation
		remove_variable = byzfm_imperial_palace_var
		set_variable = byzfm_mega_palation_complete
		trigger_event = {
			id = byzfm.101 # Imperial Palace Ruins Identified
		}
	}

	invalid = {
		NOT = {
			owns_entire_state_region = STATE_EASTERN_THRACE
		}
	}

	on_invalid = {
		remove_modifier = modifier_grefm_active_excavation
		remove_variable = byzfm_imperial_palace_var
	}

	current_value = {
		value = root.var:byzfm_imperial_palace_var
	}

	goal_add_value = {
		value = 24
	}

	progressbar = yes

	weight = 100

	should_be_pinned_by_default = yes
	
}

je_byzfm_hippodrome_survey = {
	icon = "gfx/interface/icons/event_icons/event_ruins.dds"
	
	group = je_group_byzfm_content

	immediate = {
		set_variable = {
			name = byzfm_hippodrome_var
			value = 0
		}
		s:STATE_EASTERN_THRACE = {
			random_scope_state = {
				save_scope_as = byzantium_capital_state
			}
		}
	}

	complete = {
		scope:journal_entry = { is_goal_complete = yes }
	}

	on_complete = {
		remove_modifier = modifier_grefm_active_excavation
		remove_variable = byzfm_hippodrome_var
		set_variable = byzfm_hippodrome_complete
		trigger_event = {
			id = byzfm.103 # hippodrome survey complete
		}
	}

	invalid = {
		NOT = {
			owns_entire_state_region = STATE_EASTERN_THRACE
		}
	}

	on_invalid = {
		remove_modifier = modifier_grefm_active_excavation
		remove_variable = byzfm_hippodrome_var
		remove_variable = byzfm_hippodrome_fallback
	}

	current_value = {
		value = root.var:byzfm_hippodrome_var
	}

	goal_add_value = {
		value = 24
	}

	on_monthly_pulse = {
		effect = {
			change_variable = {
				name = byzfm_hippodrome_var
				add = 1
			}
		}
	}

	progressbar = yes

	weight = 10

	should_be_pinned_by_default = yes
	
}
