je_balkfm_nacertanije = {
	icon = "gfx/interface/icons/event_icons/event_flag_yug.dds"
	
	group = je_group_balkfm_yugoslavism
	
	should_be_pinned_by_default = yes
	
	is_shown_when_inactive = {
		OR = {
			country_has_primary_culture = cu:serb
			country_has_primary_culture = cu:croat
			country_has_primary_culture = cu:bosniak
			country_has_primary_culture = cu:slovene
			country_has_primary_culture = cu:bulgarian
		}
		NOR = { 
			has_global_variable = has_formed_yug
			has_global_variable = balkfm_yugoslavism_failed
		}
		trigger_if = { # BUL gets access to this journal via option in balkfm.585
			limit = {
				country_has_primary_culture = cu:bulgarian
			}
			has_variable = balkfm_texas_left
		}
	}
	
	possible = {
		has_technology_researched = nationalism
		is_subject = no
		hidden_trigger = { # blocks journal before a tag has independence
			is_secessionist = no
		}
	}
	
	immediate = {
		trigger_event = {
			id = balkfm.200 # Yugoslavism Starter Event
		}
	}
	
	scripted_button = balkfm_form_yugoslavia_button
	scripted_button = balkfm_austrian_agitation_button
	scripted_button = balkfm_ottoman_agitation_button
	
	complete = {
		custom_tooltip = {
			text = je_balkfm_nacertanije_complete_tt
			hidden_trigger = {
				c:YUG ?= this
			}
		}
		trigger_if = {
			limit = {
				has_variable = balkfm_texas_left
			}
			owns_entire_state_region = STATE_BULGARIA
			owns_entire_state_region = STATE_NORTHERN_THRACE
			owns_entire_state_region = STATE_DOBRUDJA
		}
	}
	
	on_complete = {
		if = {
			limit = {
				has_variable = balkfm_texas_left
			}
			trigger_event = {
				id = balkfm.587 # The Slavic Balkans
			}
		}
		else = {
			trigger_event = {
				id = balkfm.209 # The Birth of Yugoslavia
			}
		}
		balkfm_cleanup_yugoslavism_effect = yes
	}
	
	fail = {
		custom_tooltip = {
			text = je_balkfm_nacertanije_fail_tt
			hidden_trigger = {
				OR = {
					c:SRPS ?= this
					c:TSAR ?= this
					c:TRIU ?= this
					c:BYZ ?= this
				}
			}
		}
	}
	
	on_fail = {
		balkfm_cleanup_yugoslavism_effect = yes
	}
	
	invalid = {
		OR = {
			custom_tooltip = {
				text = balkfm_yugoslavism_completed_tt
				hidden_trigger = {
					has_global_variable = has_formed_yug
					NOT = {
						c:YUG ?= this
					}
				}
			}
			custom_tooltip = {
				text = balkfm_yugoslavism_failed_tt
				hidden_trigger = {
					has_global_variable = balkfm_yugoslavism_failed
				}
			}
			custom_tooltip = {
				text = balkfm_weaksauce_trigger_tt
				hidden_trigger = {
					has_variable = balkfm_weaksauce
				}	
			}
		}
	}
	
	on_invalid = {
		balkfm_cleanup_yugoslavism_effect = yes
	}
	
	on_weekly_pulse = {
		effect = {
			balkfm_determine_yugoslav_leader_effect = yes
		}
		events = {
#			balkfm.204 # Yugoslav Uprising in X
		}
	}
	
	on_monthly_pulse = {
		random_events = {
			180 = 0
			20 = balkfm.206 # Yugoslav Breakaways Seek Union!
			5 = balkfm.207 # Yugoslav Annexation
		}
	}
	
	status_desc = {
		triggered_desc = {
			desc = je_balkfm_nacertanije_status
		}
	}

	weight = 2000
	
}

je_balkfm_mon_black_mountain_new = {
# This journal is unfortunately more convoluted than I would like and uses a system of global variables to ensure values are passed between MON and TUR
# Next province variables are handled in root scopes to simplify scripting
# Monthly progress variables are stored locally
# The following global variables are used:
	## balkfm_border_influence_var -> stores the value for the journal progressbar
	## balkfm_border_incursion_var -> 0 = no incursion, 1 = MON incursion, 2 = TUR incursion
	## balkfm_mon_border_control_var -> stores the number of provinces controled by MON
	## balkfm_tur_border_control_var -> stores the number of provinces controled by TUR
	## balkfm_mon_next_province_var -> stores the next province for MON to claim
	## balkfm_tur_next_province_var -> stores the next province for TUR to claim
# The following local variables are used:
	## balkfm_border_control_var -> tracks number of controlled provinces
	## balkfm_mon_border_progress_var -> cached for performance
	## balkfm_tur_border_progress_var -> cached for performance
	## balkfm_total_border_progress_var -> cached for performance
	
	icon = "gfx/interface/icons/event_icons/event_montenegro.dds"
	
	group = je_group_balkfm_montenegro
	
	should_be_pinned_by_default = yes
	
	immediate = {
		add_modifier = modifier_balkfm_practicing_headtaking
		set_variable = {
			name = balkfm_view_captives_var
			value = 0
		}
		balkfm_update_progress_variables_effect = yes
	}
	
	modifiers_while_active = {
		modifier_balkfm_border_feud
	}
	
	scripted_progress_bar = balkfm_mon_border_progress_bar
	
	scripted_button = balkfm_reinforce_border_button
	scripted_button = balkfm_border_raid_button
	scripted_button = balkfm_view_captives_button
	scripted_button = balkfm_mon_claim_land_button
	scripted_button = balkfm_annex_tribe_button
	scripted_button = balkfm_renounce_headtaking_button
	scripted_button = balkfm_negociate_kotor_button	
	
	complete = {
		custom_tooltip = {
			text = je_balkfm_mon_black_mountain_complete_tt
			hidden_trigger = {
				balkfm_border_control_value >= 5
			}
		}
	}
	
	on_complete = {
		trigger_event = {
			id = balkfm.369 # The Lands of Montenegro
		}
		balkfm_border_cleanup_effect = yes
		set_global_variable = balkfm_border_conflict_resolved
	}
	
	fail = {
		OR = {
			custom_tooltip = {
				text = je_balkfm_mon_black_mountain_fail_tt
				hidden_trigger = {
					NOT = {
						any_scope_state = {
							state_region = s:STATE_MONTENEGRO
						}
					}
				}
			}
		}
		trigger_if = {
			limit = {
				exists = c:TUR
			}
			OR = {
				is_subject_of = c:TUR
				is_in_same_power_bloc = c:TUR
			}
		}
	}
	
	on_fail = {
		trigger_event = {
			id = balkfm.368 # International Indifference
		}
		balkfm_border_cleanup_effect = yes
		set_global_variable = balkfm_border_conflict_resolved
	}
	
	invalid = {
		hidden_trigger = {
			has_global_variable = balkfm_border_conflict_resolved
		}
	}
	
	on_invalid = {
		balkfm_border_cleanup_effect = yes
	}
	
	on_weekly_pulse = {
		effect = {
			balkfm_mon_determine_next_province_effect = yes
			balkfm_update_captive_variables_effect = yes
			balkfm_update_progress_variables_effect = yes
		}
	}
	
	on_monthly_pulse = {
		effect = {
			change_global_variable = {
				name = balkfm_border_influence_var
				add = var:balkfm_total_border_progress_var
			}
			clamp_global_variable = {
				name = balkfm_border_influence_var
				max = 1000
				min = 0
			}
			balkfm_update_border_progressbar_effect = yes
		}
		events = {
			balkfm.355 # Montenegrin Tribe Rebels
		}
		random_events = {
			200 = 0
			20 = balkfm.348 # Country Crosses the Border (Attacker)
			5 = balkfm.341 # Border Villages Burned (Defender Event)
			5 = balkfm.342 # Village Skirmish Escalates (Defender Event)
			5 = balkfm.343 # Border Incursion Repelled (Defender Event)
			5 = balkfm.356 # Tribe Evades Taxation/Requests Aid
		}
	}
	
	status_desc = {
		triggered_desc = {
			desc = je_balkfm_mon_black_mountain_status
		}
	}
	
	weight = 1000
	
}
