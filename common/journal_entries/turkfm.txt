	}
	
	scripted_progress_bar = turkfm_tanzimat_progress_bar
	
	scripted_button = turkfm_start_tanzimat_button
	
	complete = {
		custom_tooltip = {
			text = je_turkfm_tanzimat_main_complete
			hidden_trigger = {
				var:turkfm_tanzimat_var >= 1000
			}
		}
	}
	
	on_complete = {
		turkfm_cleanup_tanzimat_effect = yes
	}
	
	fail = {
		custom_tooltip = {
			text = je_turkfm_tanzimat_main_fail
			hidden_trigger = {
				var:turkfm_tanzimat_var <= 0
			}
		}
	}
	
	on_fail = {
		turkfm_cleanup_tanzimat_effect = yes
	}
	
	on_weekly_pulse = {
		effect = {
			set_variable = {
				name = turkfm_tanzimat_progress_var
				value = turkfm_tanzimat_progress_value
			}
		}
	}
	
	on_monthly_pulse = {
		random_events = {
			chance_of_no_event = 95
			5 = turkfm.102 # The Men of the Tanzimat
		}
		effect = {
			change_variable = {
				name = turkfm_tanzimat_var
				add = var:turkfm_tanzimat_progress_var
			}
			clamp_global_variable = {
				name = turkfm_tanzimat_var
				max = 1000
				min = 0
			}
			je:je_turkfm_tanzimat_main = {
				set_bar_progress = {
					name = turkfm_tanzimat_progress_bar
					value = var:turkfm_tanzimat_var
				}
			}
		}
	}
	
	weight = 4000

}
