# GENERAL TRIGGERS #

balkfm_balkans_involved_trigger = {
	OR = {
		has_interest_marker_in_region = sr:region_balkans
		capital = {
			state_in_balkans = yes
		}
	}
}

balkfm_balkans_native_trigger = {
	AND = {
		OR = {
			country_has_primary_culture = cu:greek
			country_has_primary_culture = cu:serb
			country_has_primary_culture = cu:croat
			country_has_primary_culture = cu:bosniak
			country_has_primary_culture = cu:slovene
			country_has_primary_culture = cu:bulgarian
		}
		capital = {
			state_in_balkans = yes
		}
	}
}

balkfm_is_national_empire_trigger = {
	OR = {
		c:HEMP ?= this
		c:BYZ ?= this
		c:SRPS ?= this
		c:TSAR ?= this
		c:TRIU ?= this
	}
}

balkfm_is_balkan_empire_trigger = {
	OR = {
		c:HEMP ?= this
		c:BYZ ?= this
		c:YUG ?= this
		c:SRPS ?= this
		c:TSAR ?= this
		c:TRIU ?= this
	}
}

balkfm_has_yugoslav_culture_trigger = {
	OR = {
		country_has_primary_culture = cu:serb
		country_has_primary_culture = cu:croat
		country_has_primary_culture = cu:bosniak
		country_has_primary_culture = cu:slovene
		country_has_primary_culture = cu:bulgarian
	}
}

balkfm_shares_slav_culture_trigger = {
	trigger_if = {
		limit = {
			country_has_primary_culture = cu:serb
		}
		any_primary_culture = {
			cu:serb = this
		}
	}
	trigger_if = {
		limit = {
			country_has_primary_culture = cu:bulgarian
		}
		any_primary_culture = {
			cu:bulgarian = this
		}
	}
	trigger_if = {
		limit = {
			country_has_primary_culture = cu:bosniak
		}
		any_primary_culture = {
			cu:bosniak = this
		}
	}
	trigger_if = {
		limit = {
			country_has_primary_culture = cu:croat
		}
		any_primary_culture = {
			cu:croat = this
		}
	}
	trigger_if = {
		limit = {
			country_has_primary_culture = cu:slovene
		}
		any_primary_culture = {
			cu:slovene = this
		}
	}
}

balkfm_is_yugoslav_homeland_trigger = {
	state_region = {
		OR = {
			is_homeland = cu:serb
			is_homeland = cu:bulgarian
			is_homeland = cu:bosniak
			is_homeland = cu:croat
			is_homeland = cu:slovene
		}
	}
}

balkfm_yugoslav_radicals_trigger = {
	trigger_if = {
		any_scope_pop = {
			has_pop_culture = serb
		}
		radical_fraction = {
			culture = cu:serb
			value >= 0.33
		}
	}
	trigger_if = {
		any_scope_pop = {
			has_pop_culture = bulgarian
		}
		radical_fraction = {
			culture = cu:bulgarian
			value >= 0.33
		}
	}
	trigger_if = {
		any_scope_pop = {
			has_pop_culture = bosniak
		}
		radical_fraction = {
			culture = cu:bosniak
			value >= 0.33
		}
	}
	trigger_if = {
		any_scope_pop = {
			has_pop_culture = croat
		}
		radical_fraction = {
			culture = cu:croat
			value >= 0.33
		}
	}
	trigger_if = {
		any_scope_pop = {
			has_pop_culture = slovene
		}
		radical_fraction = {
			culture = cu:slovene
			value >= 0.33
		}
	}
}
