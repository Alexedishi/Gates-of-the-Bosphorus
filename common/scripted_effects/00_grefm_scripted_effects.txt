purge_turkish_homelands = {
	every_scope_state = {
		limit = {
			AND = {
				NOT = { has_variable = grefm_turkish_reserve }
				state_region = {
					is_homeland = cu:turkish
				}
			}
		}
		add_radicals_in_state = {
			culture = cu:turkish
			value = small_radicals
		}
		state_region = {
			remove_homeland = turkish
		}
	}
}

recuit_tzavelas = {

		limit = {

		}
		save_scope_as = tzavelas_scope
	}
	scope:tzavelas_scope = {
		transfer_character = c:GRE
		set_as_interest_group_leader = yes
	}
