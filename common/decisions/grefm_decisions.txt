grefm_corinth_canal_survey_decision = {
	is_shown = {
		owns_entire_state_region = STATE_PELOPONNESE
		NOR = {
			any_scope_state = {
				has_building = building_corinth_canal
			}
			has_variable = grefm_corinth_canal_var
			has_variable = grefm_corinth_survey_complete
		}
	}
	
	possible = {	
		produced_bureaucracy > 200
		owns_entire_state_region = STATE_PELOPONNESE
		has_technology_researched = railways
		NOT = { has_journal_entry = je_grefm_corinth_canal_survey }
	}
	
	when_taken = {
		set_variable = { 
			name = grefm_corinth_canal_var
			value = 0
		}
		add_journal_entry = {
			type = je_grefm_corinth_canal_survey
		}
	}
	
	ai_chance = {
		base = 0
		modifier = {
			trigger = { produced_bureaucracy > 500 }
			add = 100
		}
		modifier = {
			OR = {
				is_diplomatic_play_committed_participant = yes
				is_at_war = yes
			}
			add = -1000
		}
	}
}

		}	
	}
}

grefm_annex_hepatanese_decision = {
	is_shown = {
		NOT = { has_variable = ionian_islands_var }
		country_has_primary_culture = cu:hellene
		is_subject = no
		exists = c:ION
		c:ION = {
			is_subject = yes
			is_player = no
			NOT = { is_subject_of = ROOT }
			top_overlord = {
				is_player = no
			}
		}
	}

	possible = {
		is_at_war = no
		has_technology_researched = nationalism
		custom_tooltip = {
			text = ionian_islands_possible_tt
			any_scope_state = {
				state_region = {
					is_homeland = cu:hellene
				}
				count >= 3
			}
		}
		c:ION = {
			relations:root >= relations_threshold:cordial
			is_at_war = no
			top_overlord = {
				relations:root >= relations_threshold:cordial
				NOT = { has_war_with = ROOT }
			}
		}
	}

	when_taken = {
		set_variable = ionian_islands_var
		custom_tooltip = treaty_of_london_event_tt
		hidden_effect = {
			trigger_event = {
				id = grefm.580 # ION Annexation
				days = 0
				popup = yes
			}
		}
	}

	ai_chance = {
		base = 10
	}
}

	is_shown = {
		any_scope_building = {
			is_building_type = building_hippodrome
		}
		NOT = {
			has_modifier = modifier_byzfm_chariot_racing
		}
		NOT = {
			has_modifier = modifier_byzfm_expanded_chariot_racing
		}
	}
	
	possible = {
		has_technology_researched = organized_sports
	}
	
	when_taken = {
		custom_tooltip = {
			text = chariot_racing_enabled_tt	
			trigger_event = {
				id = byzfm.121
				popup = yes
			}
		}
		show_as_tooltip = {
			add_modifier = modifier_byzfm_chariot_racing
		}
	}
}

grefm_expand_chariot_racing_decision = {
	is_shown = {
		has_modifier = modifier_byzfm_chariot_racing
		NOT = {
			has_modifier = modifier_byzfm_expanded_chariot_racing
		}
	}
	
	possible = {
		has_modifier = modifier_byzfm_chariot_racing
		has_technology_researched = electric_telegraph
	}
	
	when_taken = {
		show_as_tooltip = {
			remove_modifier = modifier_byzfm_chariot_racing
			add_modifier = {
				name = modifier_byzfm_expanded_chariot_racing
			}
		}
		hidden_effect = {
			trigger_event = {
				id = byzfm.122
			}
		}
	}
