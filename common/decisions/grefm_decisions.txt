grefm_corinth_canal_decision = {
	is_shown = {
		has_technology_researched = railways
		has_variable = grefm_1843_revolution_happened
		NOT = {
			any_scope_building = {
				is_building_type = building_corinth_canal
			}
		}
		NOT = {
			any_scope_state = {
				has_modifier = modifier_grefm_corinth_canal_survey
			}
		}
		NOT = {
			has_variable = grefm_corinth_canal_var
		}
	}
	
	possible = {	
		produced_bureaucracy > 500
		owns_entire_state_region = STATE_PELOPONNESE
	}
	
	when_taken = {
		set_variable = { 
			name = grefm_corinth_canal_var
			value = 0
		}
		add_modifier = {
			name = modifier_grefm_corinth_canal_survey
			months = 24
		}
		add_journal_entry = {
			type = je_grefm_corinth_canal_survey
		}
	}
	
	ai_chance = {
		base = 0
		modifier = {
			trigger = { produced_bureaucracy > 2500 }
			add = 5
		}	
		modifier = {
			trigger = { 
				produced_bureaucracy > 5000 
				any_country = {
					is_ai = no
					has_variable = grefm_corinth_canal_var
				}
			}
			add = 50
		}
		modifier = {
			OR = {
				is_diplomatic_play_committed_participant = yes
				is_at_war = yes
			}
			add = -1000
		}	
	}
}

grefm_dismiss_bavarians_decision = {
	is_shown = {
		NOT = { has_variable = grefm_dismissed_daddys_boys }
		has_modifier = modifier_grefm_bavarian_auxiliaries
	}
	
	possible = {
		has_modifier = modifier_grefm_bavarian_auxiliaries
	}
	
	when_taken = {
		show_as_tooltip = {
			remove_modifier = modifier_grefm_bavarian_auxiliaries
			remove_modifier = modifier_grefm_funding_bavarian_auxiliaries
			add_modifier = {
				name = modifier_grefm_dismissed_bavarians
				months = normal_modifier_time
				is_decaying = yes
			}
		}
		hidden_effect = {
			trigger_event = {
				id = grefm.018 # Bavarian Auxiliaries Dismissed
				popup = yes
			}
		}
		set_variable = grefm_dismissed_daddys_boys
	}
	
	ai_chance = {
		base = 1000 # AI Greece just dies with so many expenses
	}
	
}

grefm_lake_kopais_decision = {
	is_shown = {
		has_variable = grefm_lake_kopais_option

		owns_entire_state_region = STATE_ATTICA
	}
	
	possible = {

	}

			name = grefm_lake_kopais_var

		}
		add_modifier = {
			name = modifier_grefm_reclaiming_land
		}
		add_journal_entry = {
			type = je_grefm_land_reclaimation
		}
	}
	
	ai_chance = {
		base = 0
		modifier = {
			trigger = { produced_bureaucracy > 2500 }
			add = 5
		}	
	}
}

grefm_annex_epirus_decision = {
	is_shown = {
		NOT = { has_variable = grefm_epirus_var }
		country_has_primary_culture = cu:greek
		is_subject = no
		exists = c:EPI
		c:EPI = {
			OR = {
				is_subject = no
				is_subject_of = ROOT
			}
			is_player = no
		}
	}

	possible = {
		is_at_war = no
		has_technology_researched = nationalism
		c:EPI = {
			relations:root >= relations_threshold:cordial
			is_at_war = no
		}
		c:EPI = {
			NOT = { has_truce_with = ROOT }
		}
	}

	when_taken = {
		set_variable = grefm_epirus_var
		custom_tooltip = treaty_of_eprius_event_tt
		hidden_effect = {
			trigger_event = {
				id = grefm.221 # EPI Annexation
				days = 0
				popup = yes
			}
		}
	}

	ai_chance = {
		base = 10
	}
}

grefm_release_turkophagos_decision = {
	is_shown = {
		NOT = { has_variable = grefm_turkeater_released }
		has_journal_entry = je_grefm_bavarocracy_main
		year <= 1849
	}

	possible = {
		year <= 1849
	}

	when_taken = {
		trigger_event = {
			id = grefm.019 # nikitas released from jail
		}
		show_as_tooltip = {
			add_modifier = {
				name = modifier_grefm_nikitas_released
				months = normal_modifier_time
				is_decaying = yes
			}
		}
		set_variable = grefm_turkeater_released
	}

	ai_chance = {
		base = 0
	}
	
}
