eocfm_UKR_decision = {
	is_shown = {
		has_journal_entry = je_eocfm_the_orthodox_realms
		NOT = { has_variable = eocfm_UKR_var }
		exists = c:UKR
		c:UKR = {
			is_subject = no
			is_player = no
		}
	}
	
	possible = {
		c:UKR = {
			relations:root >= relations_threshold:cordial
			is_at_war = no
			top_overlord = {
				relations:root >= relations_threshold:cordial
				NOT = { has_war_with = ROOT }
			}
		}
	}
	
	when_taken = {
		set_variable = eocfm_UKR_var
		trigger_event = {
			id = eocfm.320 # The Ukranian Domain
		}
	}

	ai_chance = {
		base = 10
	}
}
