	type = country
	
	text = {
		trigger = {
			any_country = {
				owns_entire_state_region = STATE_ALBANIA
				grefm_is_muslim_trigger = yes
			}
			any_country = {
				has_modifier = modifier_grefm_supporting_epirus
			}
		}
		localization_key = custom_grefm_epirus_revolt

		}
		localization_key = custom_grefm_samos_progress
	}
