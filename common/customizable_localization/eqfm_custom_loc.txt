## Cultural Tension Custom Loc ##

eqfm_greek_tension_type = {
	type = country
	
	text = {
		trigger = {
			any_scope_state = {
				state_in_balkans = yes
				state_region = {
					is_homeland = cu:greek
				}
			}
		}
		localization_key = custom_eqfm_greek_tension
	}
}

## Boost Values Custom Loc ##


eqfm_nationalism_tension_type = {
	type = country
	
	text = {
		trigger = {
			any_country = {
				balkfm_balkans_native_trigger = yes
				has_technology_researched = nationalism
			}
		}
		localization_key = custom_eqfm_nationalism_tension
	}
}

eqfm_accelerate_tension_type = {
	type = country
	
	text = {
		trigger = {
			any_country = {
				has_modifier = modifier_eqfm_accelerating_tensions
			}
		}
		localization_key = custom_eqfm_accelerate_tension
	}
}
